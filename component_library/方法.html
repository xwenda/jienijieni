// promise：请求方法，success：成功回调，fail：失败回调
export const handleRequestResponseTips = async (promise, success, fail) => {
  return new Promise(async (resolve, reject) => {
    const res = await promise.catch((err) => {
      fail && fail(err);
      reject();
    });
    if (res.code === 0) {
      ElMessage.success({
        message: res.msg || (res.data === String && res.data) || "操作成功",
        duration: 2000,
      });
      success && success(res);
      resolve();
    } else {
      ElMessage.error({
        message: res.msg || (res.data === String && res.data) || "操作失败",
        duration: 2000,
      });
      fail && fail();
      reject();
    }
  });
};

// timestamp：数组，包含两个对比时间，returnString：布尔值，是否返回字符串
export const handleCutTime = (timestamp, returnString) => {
  if (Array.isArray(timestamp)) {
    timestamp = Math.abs(
      new Date(timestamp[0]).getTime() - new Date(timestamp[1]).getTime()
    );
  }
  let day = parseInt(timestamp / 86400000);
  day = day >= 10 ? day : `0${day}`;
  let hour = parseInt((timestamp % 86400000) / 3600000);
  hour = hour >= 10 ? hour : `0${hour}`;
  let minutes = parseInt(((timestamp % 86400000) % 3600000) / 60000);
  minutes = minutes >= 10 ? minutes : `0${minutes}`;
  let second = parseInt((((timestamp % 86400000) % 3600000) % 60000) / 1000);
  second = second >= 10 ? second : `0${second}`;
  return returnString
    ? `${day ? day + "d" : ""}${hour ? hour + "h" : ""}${
        minutes ? minutes + "m" : ""
      }${second ? second + "s" : ""}`
    : { day, hour, minutes, second };
};

// time：任意时间，type: 1-季度、2-周、3-半年
export const handleTimeByType = (time, type) => {
  if (!time) return [null, null];
  const specifiedDate = moment(time);
  if (type === 1) {
    return [
      specifiedDate.clone().startOf("quarter").format("YYYY-MM-DD 00:00:00"),
      specifiedDate.clone().endOf("quarter").format("YYYY-MM-DD 23:59:59"),
    ];
  } else if (type === 2) {
    return [
      specifiedDate.startOf("isoWeek").format("YYYY-MM-DD 00:00:00"),
      specifiedDate.endOf("isoWeek").format("YYYY-MM-DD 23:59:59"),
    ];
  } else {
    // 获取指定日期所在半年的开始日期和结束日期
    let startOfHalfYear, endOfHalfYear;
    if (specifiedDate.month() < 6) {
      // 上半年
      startOfHalfYear = specifiedDate.clone().startOf("year"); // 获取上半年的开始日期
      endOfHalfYear = specifiedDate
        .clone()
        .startOf("year")
        .add(5, "months")
        .endOf("month"); // 获取上半年的结束日期
    } else {
      // 下半年
      startOfHalfYear = specifiedDate.clone().startOf("year").add(6, "months"); // 获取下半年的开始日期
      endOfHalfYear = specifiedDate.clone().endOf("year"); // 获取下半年的结束日期
    }

    // 格式化日期
    const formattedStartOfHalfYear = startOfHalfYear.format(
      "YYYY-MM-DD 00:00:00"
    ); // 格式化半年的开始日期
    const formattedEndOfHalfYear = endOfHalfYear.format("YYYY-MM-DD 23:59:59"); // 格式化半年的结束日期
    return [formattedStartOfHalfYear, formattedEndOfHalfYear];
  }
};

    return target;
  }
  if (select[0].hasOwnProperty("dictValue") && !value) {
    value = "dictValue";
    name = "dictName";

  if (select[0].hasOwnProperty("value") && !value) {
    value = "value";
    name = "label";

function hexToRGB(hex) {
  let rgb = [];
  if (/^\#[0-9A-F]{3}$/i.test(hex)) {
    let sixHex = "#";
    hex.replace(/[0-9A-F]/gi, function (kw) {
      sixHex += kw + kw;
    });
    hex = sixHex;
  }
  if (/^\#[0-9A-F]{6}$/i.test(hex)) {
    hex.replace(/[0-9A-F]{2}/gi, function (kw) {
      rgb.push(eval("0x" + kw));
    });
    return `rgb(${rgb.join(",")})`;
  } else {
    console.log("颜色错误");
    return "rgb(0,0,0)";
  }
}

function rgb2Hex(rgb) {
  if (/^rgb\((\d{1,3}\,){2}\d{1,3}\)$/i.test(rgb)) {
    let hex = "#";
    rgb.replace(/\d{1,3}/g, function (kw) {
      kw = parseInt(kw).toString(16);
      kw = kw.length < 2 ? 0 + kw : kw;
      hex += kw;
    });
    return hex;

    console.log("颜色错误");
    return "#000";
  }
}

export function MixExtends(...mixins) {
  class Mix {
    constructor() {
      for (let mixin of mixins) {
        Object.assign(this, new mixin());
      }
    }
  }

  for (let mixin of mixins) {
    copyProperties(Mix, mixin);
    copyProperties(Mix.prototype, mixin.prototype);
  }

  // 将源对象的方法复制到目标对象中
  function copyProperties(target, source) {
    for (let key of Reflect.ownKeys(source)) {
      if (key !== "constructor" && key !== "prototype" && key !== "name") {
        let desc = Object.getOwnPropertyDescriptor(source, key);
        Object.defineProperty(target, key, desc);
      }
    }
  }
  return Mix;
}

export const parseVNodeV2 = (vnode) => {
  let rdom = null;
  rdom = document.createElement(vnode.tag);
  // 元素
  let attrs = vnode.data;
  if (vnode.data) {
    for (let key in attrs) rdom.setAttribute(key, attrs[key]);
    if (vnode.data.attrs) {
      for (let key in vnode.data.attrs)
        rdom.setAttribute(key, vnode.data.attrs[key]);
    }
    if (vnode.data.on) {
      for (let key in vnode.data.on) rdom[`on${key}`] = vnode.data.on[key];
    }
  }
  let children = vnode.children;
  if (children) {
    for (let i = 0; i < children.length; i++) {
      rdom.appendChild(parseVNode(children[i]));
    }
  }

  if (vnode.text) rdom = document.createTextNode(vnode.text);
  return rdom;
};

export const parseVNodeV3 = (vnode) => {
  let rdom = null;
  rdom = document.createElement(vnode.type);
  // 元素
  let attrs = vnode.props;
  if (vnode.props) {
    for (let key in attrs) {
      if (key.includes("on")) {
        rdom[key.toLowerCase()] = attrs[key];
      } else {
        rdom.setAttribute(key, attrs[key]);
      }
    }
  }
  let children = vnode.children;
  if (children) {
    for (let i = 0; i < children.length; i++) {
      if (typeof children[i] === "string") {
        rdom.innerText += children[i];
      } else {
        rdom.appendChild(parseVNode(children[i]));
      }
    }
  }
  return rdom;
};

export const getRenderV2 = (data, component) => {
  const uuid = "RENDER_MAP_POPOVER_" + getUUID();
  const wrap = document.createElement("div");
  const content = document.createElement("div");
  const cache = { observer: null, vm: null };
  wrap.setAttribute("id", uuid);
  wrap.appendChild(content);
  if (!component.mixins) component.mixins = [];
  component.mixins.push({
    methods: {
      getPoint() {
        return this.$parent.RENDER_DATA;
      },
    },
  });
  cache.vm = new Vue({
    el: content,
    data: {
      RENDER_DATA: data,
    },
    render: (h) => h(component),
  });

  function check() {
    const target = document.getElementById(uuid);
    if (!target) {
      cache.vm.$destroy();
      cache.observer.disconnect();
      cache.vm = null;
      cache.observer = null;
    }
  }

  function observer() {
    // 此处无需清除setTimeout，一定要执行
    setTimeout(() => {
      const parent = wrap.parentNode;
      check();
      if (parent && MutationObserver) {
        cache.observer = new MutationObserver(() => check());
        cache.observer.observe(document.body, {
          childList: true,
          subtree: true,
        });
      }
    }, 500);
  }

  observer();

  return wrap;
};
