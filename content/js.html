防抖（触发最后一次） ​

js
function debounce(fn,wait = 1000,immediate = false) {
	let timer = null
	return function debounced(...args) {
		if (timer) clearTimeout(timer)

		if(immediate && !timer) {
			fn.apply(this,...args)
			timer = setTimeout(() => {
				timer = null
			}, wait)
			return
		}

		timer = setTimeout(() => {
			fn.apply(this, ...args)
			timer = null
		}, wait)
	}
}

function cloneType(target){
	return Object.prototype.toString.call(target).slice(8,-1)
}

function cloneDeep(target){
	let result
	let cloneTypeResult = cloneType(target)
	if(cloneTypeResult === 'Object'){
		result = {}
	} else if(cloneTypeResult === 'Array'){
		result = []
	} else{
		result = target
	}

	for(let i in target){
		let item = target[i]
		let itemType = cloneType(item)
		if(itemType === 'Object' || itemType === 'Array'){
			result[i] = cloneDeep(item)
		}else{
			result[i] = item
		}
	}
	return result
}

class Promise {
    constructor(runPromise) {
        this.state = 'pending'
        this.successVal
        this.failReason
        this.onResolvedCallbacks = []
        this.onRejectedCallbacks = []

            this.state = 'fulfilled'
            this.successVal = val
            this.onResolvedCallbacks.forEach(fn => fn())
        }
        let reject = (reason) => {
            this.state = 'rejected'
            this.failReason = reason
            this.onRejectedCallbacks.forEach(fn => fn())
        }

        try {
            runPromise(resolve, reject)
        } catch (err) {
            reject(err)
        }
    }


    then(onFulfilled, onRejected) {
        onFulfilled = typeof onFulfilled === 'function' ? onFulfilled : value => value
        onRejected = typeof onRejected === 'function' ? onRejected : err => { throw err }
        let promise2 = new Promise((resolve, reject) => {
            if (this.state == 'fulfilled') {
                setTimeout(() => {
                    try {
                        let x = onFulfilled(this.successVal)
                        resolvePromise(promise2, x, resolve, reject)
                    } catch (err) {
                        reject(err)
                    }
                })

                        reject(err)

Promise.race = function (promiseArr) {

说一下对Promise的了解 ​

  • 有三个状态，pending、fulfilled、rejected
  • 状态改变后不可逆
  • 可链式调用
  • 内部保存一个值

Promise的all和race有什么区别 ​

all：

  • 接收一个Iterator结构的数据作为参数，其中的每个值都是promise实例
  • 只有状态都变成fulfilled才，all的状态才会变成fulfilled
  • 只有所有值中均没有reject，才会进入all的catch

race：

  • 和all相似，接收一个Iterator结构的数据作为参数，其中的每个值都是promise实例
  • 其中一个实例改变状态，race状态就跟着改变

堆和栈的区别 ​

栈(stack)：

  • 自动释放
  • 存储一些基础变量(string、number、boolean、null、undefined)以及一些对象的引用变量
  • 固定大小
  • 线性结构，后进先出
  • 读取速度快

堆(heap)：

  • 大小不固定
  • 不会自动释放，一般由程序员分配释放，也可以由垃圾回收机制回收
  • 存放function、array、object

instanceof的实现原理 ​

用于检测构造函数的 prototype属性是否是某个实例对象的原型链上

实现原理：不断遍历左边值的隐式属性 __proto__，判断其是否等于右边值的 prototype，直至等于 null为止

new的实现原理 ​

new

创建一个空对象obj 将obj的原型链上的__proto__指向构造函数的原型prototype 并且在原型链__proto__上设置构造函数的constructor为要实例化的fn 传入参数，并让构造函数的fn指向obj并执行 如果fn执行后返回的是对象类型(null除外)，则返回该对象，否则返回obj

function New(fn) {
	let obj = {}
	let arg = Array.prototype.slice.call(arguments,1)
	obj.__proto__ = fn.prototype
	obj.__proto__.constructor = fn
	let ret = fn.apply(obj,arg)
	return typeof ret === 'Object' ? ret || obj : obj
}

js运行机制和事件循环 ​

js其重要用途是用于交互，为了避免出现矛盾的交互情况，所以其只能是单线程语言，web worker允许js脚本创建线程，但子线程完全受主线程控制，且不得操作dom

js 的任务分为 同步任务和 异步任务，其中 异步任务进入 任务队列(task queue)

事件循环：

  1. 所有同步任务在主线程执行，形成一个执行栈
  2. 主线程外存在一个任务队列，异步任务有了结果，就在任务队列中放置一个事件
  3. 执行栈中的任务执行完成后，系统会读取任务队列中的事件，根据事件读取对应的异步任务，将对应的异步任务结束等待状态，进入执行栈，执行任务。
  4. 主线程不断执行第三步
  • 任务队列中的事件除了I/O还包括交互时间中制定了回调函数的事件
  • 任务队列是一个先进先出的结构，但是计时器需要先检查执行时间

nodejs的事件循环 ​

  • process.nextTick：将指定任务排到所有异步任务之前，运行效率高，不用检查任务队列
  • setImmedidate：将制定的任务排到下一次事件循环执行

onload 和 DOMContentLoaded的区别 ​

  • onload：在文档中的所有对象都在DOM中，所有图片，脚本，链接以及子框都完成了装载。
  • DOMContentLoaded：无需等待样式表、图像和子框架的完全加载

async function downloadFile(fileUrl, fileName = '未命名的文件') {

function getBlob(fileUrl) {
    return new Promise(resolve => {
        const xhr = new XMLHttpRequest();
        xhr.open('GET', fileUrl, true);
        // 监听进度事件，需要在请求发送前监听
        xhr.addEventListener(
            'progress',
            function (evt) {
                if (evt.lengthComputable) {
                    const percentComplete = evt.loaded / evt.total;
                    // percentage是当前下载进度，可根据自己的需求自行处理
                    const percentage = percentComplete * 100;
                    console.log(percentage)
                }
            },
            false
        );
        xhr.responseType = 'blob';
        xhr.onload = () => {
            if (xhr.status === 200) {
                resolve(xhr.response);
            }
        };
        xhr.send();
    });
}
// 下载文件
async function downloadFileByBlog(url, filename = '未命名的文件', extensionName = 'xlsx') {
    const blob = new Blob([url], {
        type: 'application/zip;application/vnd.openxmlformats-officedocument.spreadsheetml.sheet;application/vnd.ms-excel;application/pdf'
    });
    const a = document.createElement('a');
    // 创建URL
    const blobUrl = window.URL.createObjectURL(blob);
    a.download = `${filename}.${extensionName}`;
    a.href = blobUrl
    document.body.appendChild(a);
    // 下载文件
    a.click();
    // 释放内存
    URL.revokeObjectURL(blobUrl);
    document.body.removeChild(a);
}

尾调用 ​

尾调用

即函数在最后一步调用函数自身

【function f() { return f() }】

必须是调用自身

注意，只有不再用到外层函数的内部变量，内层函数的调用帧才会取代外层函数的调用帧，否则就无法进行“尾调用优化”，若使用了内部变量，则形成了闭包

function tco(f) {
  var value;
  var active = false;
  var accumulated = [];
  return function accumulator() {
    accumulated.push(arguments);
    if (!active) {
      active = true;
      while (accumulated.length) {
        value = f.apply(this, accumulated.shift());
      }
      active = false;
      return value;
    }
  };
}

var sum = tco(function(x, y) {
